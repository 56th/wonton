245 struct Spherical3DCoordinates {
246  public:
247 
249  static constexpr double geometry_factor = 1;
250 
252  template<int D>
253  static void verify_coordinate_system() {
254  // Valid only in 3D
255  static_assert(D == 3,
256  "Spherical (3D) coordinates only valid in 3D.");
257  }
258 
260  template<int D>
261  static Vector<D> modify_gradient(Vector<D> const & gradient,
262  Point<D> const & reference_point) {
263  auto new_gradient = gradient;
264  new_gradient[1] /= reference_point[0];
265  new_gradient[2] /= (reference_point[0] * sin(reference_point[1]));
266  return(std::move(new_gradient));
267  }
268 
270  template<int D>
271  static Vector<D> modify_line_element(Vector<D> const & line_element,
272  Point<D> const & reference_point) {
273  auto new_line_element = line_element;
274  new_line_element[1] *= reference_point[0];
275  new_line_element[2] *= (reference_point[0] * sin(reference_point[1]));
276  return(std::move(new_line_element));
277  }
278 }; // Spherical (3D) Coordinates

222 // ____________________________________________________________________________
223 // Get lower and upper corners of cell bounding box
224 template<int D>
225 void Direct_Product_Mesh_Wrapper<D>::cell_get_bounds(
226  const int id, Point<D> *plo, Point<D> *phi) const {
227  std::array<int,D> indices = cellid_to_indices(id);
228  // Cell axis points are zero-indexed, cells are zero-indexed. Thus cell 0 is
229  // bounded by axis points 0 and 1, or more generally, cell N is bounded by
230  // axis points N and N+1.
231  for (int d = 0; d < D; ++d) {
232  (*plo)[d] = mesh_.get_axis_point(d, indices[d]);
233  (*phi)[d] = mesh_.get_axis_point(d, indices[d]+1);
234  }
235 }

240 // ____________________________________________________________________________
241 // Convert from indices to ID
242 template<int D>
243 int Direct_Product_Mesh_Wrapper<D>::indices_to_cellid(
244  const std::array<int,D>& indices) const {
245  assert(D == mesh_.space_dimension());
246  int id = 0;
247  // Loop over all but the last dimension
248  for (int d = D-1; d > 0; --d) {
249  int idx = indices[d];
250  int mult = axis_num_cells(d-1);
251  id += idx;
252  id *= mult;
253  }
254  // Handle the last dimension
255  id += indices[0];
256  // Return
257  return id;
258 }

260 // ____________________________________________________________________________
261 // Convert from ID to indices
262 template<int D>
263 std::array<int,D> Direct_Product_Mesh_Wrapper<D>::cellid_to_indices(
264  const int id) const {
265  assert(D == mesh_.space_dimension());
266  std::array<int,D> indices;
267  int residual = id;
268  // Construct the denominators
269  std::array<int,D> denom;
270  denom[0] = 1;
271  for (int d = 1; d < D; ++d) {
272  denom[d] = denom[d-1] * axis_num_cells(d-1);
273  }
274  // Loop over all but the last dimension
275  for (int d = D-1; d > 0; --d) {
276  int index = residual / denom[d];
277  residual -= index * denom[d];
278  indices[d] = (int) index;
279  }
280  // Handle the last dimension
281  indices[0] = (int) residual;
282  // Return
283  return std::move(indices);
284 }
